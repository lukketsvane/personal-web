import { Canvas } from "@react-three/fiber"
import { OrbitControls, useGLTF, Environment } from "@react-three/drei"
import Image from "next/image"

function Model({ url }) {
  const { scene } = useGLTF(url)
  return <primitive object={scene} />
}

## Utforsking av Form: Geometriske Studiar

Ein utforskingsreise gjennom **spomenik-inspirert parametrisk design**, der vi kombinerer moderne teknologiar med klassiske arkitektoniske prinsipp for å skape unike former og strukturar.
